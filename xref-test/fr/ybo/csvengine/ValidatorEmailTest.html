19  import static org.junit.Assert.assertEquals;
20  import static org.junit.Assert.assertNotNull;
21  import static org.junit.Assert.assertNull;
22  import static org.junit.Assert.assertTrue;

24  import fr.ybo.csvengine.model.*;
25  import fr.ybo.csvengine.model.Error;
26  import org.junit.Test;
27  
28  import fr.ybo.csvengine.annotation.CsvColumn;
29  import fr.ybo.csvengine.annotation.CsvFile;
30  import fr.ybo.csvengine.annotation.CsvValidation;
31  import fr.ybo.csvengine.exception.CsvErrorsExceededException;
32  import fr.ybo.csvengine.validator.ValidatorEmail;
33  
34  /**
35   * Test of {@link ValidatorEmail}.
36   */
37  public class ValidatorEmailTest {
38  
39  	@CsvFile
40  	public static class ObjetEmail {
41  		@CsvColumn("att")
42  		public String att;
43  
44  		@CsvValidation(ValidatorEmail.class)
45  		@CsvColumn("email")
46  		public String email;
47  	}
48  
49  	@Test
50  	public void testEmail() throws CsvErrorsExceededException {
51  		StringStream stream =
52  				new StringStream("att,email\n" + ",\n" + ",nonvalide@tutu\n" + ",nonvalide\n" + ",@nonvalide.fr\n"
53  						+ ",valide@valide.fr");
54  		CsvEngine engine =
55  				new CsvEngine(EngineParameters.createBuilder().setNbLinesWithErrorsToStop(999).build(), ObjetEmail.class);
56  		Result<ObjetEmail> result = engine.parseInputStream(stream, ObjetEmail.class);
57  		assertNotNull(result);
58  		assertEquals(2, result.getObjects().size());
59  		assertNull(result.getObjects().get(0).email);
60  		assertEquals("valide@valide.fr", result.getObjects().get(1).email);
61  
62  		assertEquals(3, result.getErrors().size());
63  		fr.ybo.csvengine.model.Error error1 = result.getErrors().get(0);
64  		assertEquals(",nonvalide@tutu", error1.getCsvLine());
65  		assertEquals(1, error1.getMessages().size());
66  		assertTrue(error1.getMessages().get(0).contains("nonvalide@tutu"));
67  
68  		Error error2 = result.getErrors().get(1);
69  		assertEquals(",nonvalide", error2.getCsvLine());
70  		assertEquals(1, error2.getMessages().size());
71  		assertTrue(error2.getMessages().get(0).contains("nonvalide"));
72  
73  		Error error3 = result.getErrors().get(2);
74  		assertEquals(",@nonvalide.fr", error3.getCsvLine());
75  		assertEquals(1, error3.getMessages().size());
76  		assertTrue(error3.getMessages().get(0).contains("@nonvalide.fr"));
77  	}
