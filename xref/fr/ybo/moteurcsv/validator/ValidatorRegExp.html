17  package fr.ybo.moteurcsv.validator;
18  
19  import java.util.Map;
20  import java.util.regex.Pattern;
21  import java.util.regex.PatternSyntaxException;
22  
23  import fr.ybo.moteurcsv.exception.InvalideParamException;
24  
25  /**
26   * Validateur permettant de valider un champ à partir d'une expression
27   * régulière. <br/>
28   * Paramètre : {@link ValidatorRegExp#PARAM_PATTERN} contient l'expression
29   * régulière ({@link Pattern}).<br/>
30   * Le paramètre format est à fournir via
31   * {@link fr.ybo.moteurcsv.annotation.BaliseCsv#params()}.
32   */
33  public class ValidatorRegExp extends ValidatorCsv {
34  
35  	/**
36  	 * Paramètre contenant l'expression régulière.
37  	 */
38  	public static final String PARAM_PATTERN = "pattern";
39  
40  	/**
41  	 * Pattern compilé.
42  	 */
43  	private Pattern pattern;
44  
45  	/*
46  	 * (non-Javadoc)
47  	 * 
48  	 * @see fr.ybo.moteurcsv.validator.ValidatorCsv#addParams(java.util.Map)
49  	 */
50  	@Override
51  	public void addParams(Map<String, String> params) throws InvalideParamException {
52  		super.addParams(params);
53  		if (!params.containsKey(PARAM_PATTERN)) {
54  			throw new InvalideParamException("Le paramètre \"" + PARAM_PATTERN + "\" est obligatoire");
55  		}
56  		try {
57  			pattern = Pattern.compile(params.get(PARAM_PATTERN));
58  		} catch (PatternSyntaxException exception) {
59  			throw new InvalideParamException("Le paramètre \"" + PARAM_PATTERN + "\" n'a pas le bon format", exception);
60  		}
61  	}
62  
63  	/* (non-Javadoc)
64  	 * @see fr.ybo.moteurcsv.validator.ValidatorCsv#validate(java.lang.String)
65  	 */
66  	@Override
67  	public void validate(String champ) throws ValidateException {
68  		if (!pattern.matcher(champ).matches()) {
69  			throw new ValidateException("La valeur " + champ + " ne correspond pas au pattern " + pattern.pattern());
70  		}
71  	}
72  
73  }
