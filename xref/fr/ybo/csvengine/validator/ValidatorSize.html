17  package fr.ybo.csvengine.validator;
18  
19  import java.util.Map;
20  
21  import fr.ybo.csvengine.exception.InvalidParamException;
22  
23  /**
24   * Validator use to validate a field on its size (min and max). <br/>
25   * Parameters :
26   * <ul>
27   * <li>{@link ValidatorSize#PARAM_MIN_SIZE} contains the minimum size of the field.</li>
28   * <li>{@link ValidatorSize#PARAM_MAX_SIZE} contains the maximum size of the field.</li>
29   * </ul>
30   * At least one of this parameters must be passed<br/>
31   * The parameters must be passed with
32   * {@link fr.ybo.csvengine.annotation.CsvValidation#params()}.<br/><br/>
33   * <p/>
34   * <u><i>French :</i></u>Validateur permettant de valider un champ sur sa taille (min et max). <br/>
35   * Paramètre :
36   * <ul>
37   * <li>{@link ValidatorSize#PARAM_MIN_SIZE} contient la taille minimum du
38   * champ.</li>
39   * <li>{@link ValidatorSize#PARAM_MAX_SIZE} contient la taille minimum du
40   * champ.</li>
41   * </ul>
42   * Au moins un des deux paramètres doit être fournit<br/>
43   * Les paramètres sont à fournir via
44   * {@link fr.ybo.csvengine.annotation.CsvValidation#params()}.
45   */
46  public class ValidatorSize extends ValidatorCsv {
47  
48      /**
49       * Parameter with the minimum size.
50       */
51      public static final String PARAM_MIN_SIZE = "minSize";
52  
53      /**
54       * Parameter with the maximum size.
55       */
56      public static final String PARAM_MAX_SIZE = "maxSize";
57  
58      /**
59       * Minimum size of the field.
60       */
61      private int tailleMin = -1;
62      /**
63       * Maximum size of the field.
64       */
65      private int tailleMax = Integer.MAX_VALUE;
66  
67      /*
68        * (non-Javadoc)
69        *
70        * @see fr.ybo.csvengine.validator.ValidatorCsv#addParams(java.util.Map)
71        */
72      @Override
73      public void addParams(Map<String, String> params) throws InvalidParamException {
74          super.addParams(params);
75          String minSizeString = params.get(PARAM_MIN_SIZE);
76          String maxSizeString = params.get(PARAM_MAX_SIZE);
77          if (minSizeString == null && maxSizeString == null) {
78              throw new InvalidParamException("At least one of the parameters \"" + PARAM_MIN_SIZE + "\" or \""
79                      + PARAM_MAX_SIZE + "\" must be passed");
80          }
81          if (minSizeString != null) {
82              try {
83                  tailleMin = Integer.parseInt(minSizeString);
84              } catch (NumberFormatException exception) {
85                  throw new InvalidParamException("The parameter \"" + PARAM_MIN_SIZE + "\" isn't in good format",
86                          exception);
87              }
88          }
89          if (maxSizeString != null) {
90              try {
91                  tailleMax = Integer.parseInt(maxSizeString);
92              } catch (NumberFormatException exception) {
93                  throw new InvalidParamException("The parameter \"" + PARAM_MAX_SIZE + "\" isn't in good format",
94                          exception);
95              }
96          }
97      }
98  
99      /*
100       * (non-Javadoc)
101       *
102       * @see fr.ybo.csvengine.validator.ValidatorCsv#validate(java.lang.String)
103       */
104     @Override
105     public void validate(String field) throws ValidateException {
106         if (field.length() < tailleMin || field.length() > tailleMax) {
107             throw new ValidateException("The value " + field + " haven't the good size (between " + tailleMin
108                     + " and " + tailleMax + ")");
109         }
110     }
