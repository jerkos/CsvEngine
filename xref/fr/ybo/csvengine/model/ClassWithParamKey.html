17  package fr.ybo.csvengine.model;
18  
19  import java.util.Arrays;
20  import java.util.HashMap;
21  import java.util.Map;
22  
23  import fr.ybo.csvengine.annotation.CsvParam;
24  
25  /**
26   * Represents a key containing the class with its parameters (use for Adapters and Validators).<br/><br/>
27   * <i><u>French :</i> Classe permettant de repréter la clé contenant une classe et ses paramètres.
28   * (utilisé pour les Adapter et la Validateur.
29   *
30   * @param <T> the class.
31   */
32  public class ClassWithParamKey<T> {
33      /**
34       * Parameters of the class.
35       */
36      private Parameter[] params;
37      /**
38       * The class.
39       */
40      private Class<? extends T> clazz;
41  
42      /**
43       * Constructor.
44       *
45       * @param params parameters.
46       * @param clazz  the class.
47       */
48      public ClassWithParamKey(CsvParam[] params, Class<? extends T> clazz) {
49          this.params = Parameter.paramsToParameters(params);
50          this.clazz = clazz;
51      }
52  
53      /**
54       * Get the parameters into a Map&lt;Name, Value&gt;.
55       *
56       * @return the resulting map.
57       */
58      public Map<String, String> getMapParams() {
59          Map<String, String> mapParams = new HashMap<String, String>();
60          for (Parameter parameter : params) {
61              mapParams.put(parameter.getName(), parameter.getValue());
62          }
63          return mapParams;
64      }
65  
66      @Override
67      public int hashCode() {
68          final int prime = 31;
69          int result = 1;
70          result = prime * result + clazz.hashCode();
71          result = prime * result + Arrays.hashCode(params);
72          return result;
73      }
74  
75      @Override
76      public boolean equals(Object obj) {
77          ClassWithParamKey<?> other = (ClassWithParamKey<?>) obj;
78          return (clazz.equals(other.clazz) && Arrays.equals(params, other.params));
79      }
80  }
