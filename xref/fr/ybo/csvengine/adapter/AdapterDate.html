17  package fr.ybo.csvengine.adapter;
18  
19  import java.text.ParseException;
20  import java.text.SimpleDateFormat;
21  import java.util.Date;
22  import java.util.Map;
23  
24  import fr.ybo.csvengine.exception.InvalidParamException;
25  import fr.ybo.csvengine.validator.ValidateException;
26  
27  /**
28   * Adapter for java.util.Date<br/>
29   * <br/>
30   * Parameter : {@link AdapterDate#PARAM_FORMAT} contain the format of the date.<br/>
31   * The parameter have to be passed with {@link fr.ybo.csvengine.annotation.CsvColumn#params()}.
32   * <br/><br/>
33   * <p/>
34   * <i><u>French :</i> Adapteur pour les dates.<br/>
35   * <br/>
36   * Paramètre : {@link AdapterDate#PARAM_FORMAT} contient le format de la date.<br/>
37   * Le paramètre format est à fournir via
38   * {@link fr.ybo.csvengine.annotation.CsvColumn#params()}.
39   */
40  public class AdapterDate extends AdapterCsv<Date> {
41  
42      /**
43       * Parameter for the Date Format.
44       */
45      public static final String PARAM_FORMAT = "format";
46  
47      /**
48       * SimpleDateFormat used for parsed and format dates.
49       */
50      private SimpleDateFormat format;
51  
52      /*
53        * (non-Javadoc)
54        *
55        * @see fr.ybo.csvengine.adapter.AdapterCsv#addParams(java.util.Map)
56        */
57      @Override
58      public void addParams(Map<String, String> params) throws InvalidParamException {
59          super.addParams(params);
60          if (!params.containsKey(PARAM_FORMAT)) {
61              throw new InvalidParamException("The parameter \"" + PARAM_FORMAT + "\" is mandatory");
62          }
63          try {
64              format = new SimpleDateFormat(params.get(PARAM_FORMAT));
65          } catch (IllegalArgumentException exception) {
66              throw new InvalidParamException("The parameter \"" + PARAM_FORMAT + "\" haven't the good format : "
67                      + params.get(PARAM_FORMAT), exception);
68          }
69      }
70  
71      /*
72        * (non-Javadoc)
73        *
74        * @see fr.ybo.csvengine.adapter.AdapterCsv#parse(java.lang.String)
75        */
76      @Override
77      public Date parse(String string) throws ValidateException {
78          try {
79              return format.parse(string);
80          } catch (ParseException e) {
81              throw new ValidateException(
82                      "A date haven't the good format, (expected format : " + format.toPattern() + ")", e);
83          }
84      }
85  
86      /*
87        * (non-Javadoc)
88        *
89        * @see fr.ybo.csvengine.adapter.AdapterCsv#toString(java.lang.Object)
90        */
91      @Override
92      public String toString(Date object) {
93          return format.format(object);
94      }
95  
96  }
