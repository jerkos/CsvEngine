34  /**
35   * Represents a column of the CSV File.<br/><br/>
36   * <i><u>French :</i> Réprésente une colonne du fichier CSV.
37   *
38   * @author ybonnel
39   */
40  public class CsvField {
41  
42      /**
43       * Adapter to use.
44       */
45      private final AdapterCsv<?> adapter;
46  
47      /**
48       * Validator to use.
49       */
50      private final List<ValidatorCsv> validators = new ArrayList<ValidatorCsv>();
51  
52      /**
53       * Map of adapters, this is useful to create just an instance per adapter.
54       */
55      private static final Map<ClassWithParamKey<AdapterCsv<?>>, AdapterCsv<?>> MAP_ADAPTERS =
56              new HashMap<ClassWithParamKey<AdapterCsv<?>>, AdapterCsv<?>>();
57  
58      /**
59       * Map of validators, this is useful to create just an instance per validator.
60       */
61      private static final Map<ClassWithParamKey<ValidatorCsv>, ValidatorCsv> MAP_VALIDATOR =
62              new HashMap<ClassWithParamKey<ValidatorCsv>, ValidatorCsv>();
63  
64      /**
65       * Field of the class to map.
66       */
67      private final Field field;
68  
69      /**
70       * True if the column is mandatory.
71       */
72      private final boolean mandatory;
73  
74      /**
75       * Constructor.
76       *
77       * @param column         annotation of the field.
78       * @param csvValidations annotation de csvValidations if it's present.
79       * @param csvValidation  annotation de csvValidation if it's present.
80       * @param field          field of the class to map.
81       */
82      public CsvField(CsvColumn column, CsvValidations csvValidations, CsvValidation csvValidation, Field field) {
83          this.field = field;
84          this.mandatory = column.mandatory();
85          this.adapter = constructAdapter(column);
86          constructValidators(csvValidations, csvValidation);
87      }
88  
89      /**
90       * Construct an adapter (reuse an existing one if the adapter with same parameters already exists).
91       *
92       * @param column annotation of the field.
93       * @return the adapter constructed.
94       */
95      private static AdapterCsv<?> constructAdapter(CsvColumn column) {
96          ClassWithParamKey<AdapterCsv<?>> key = new ClassWithParamKey<AdapterCsv<?>>(column.params(), column.adapter());
97          if (!MAP_ADAPTERS.containsKey(key)) {
98              try {
99                  Constructor<? extends AdapterCsv<?>> constructor = column.adapter().getConstructor((Class<?>[]) null);
100                 AdapterCsv<?> adapter = constructor.newInstance((Object[]) null);
101                 adapter.addParams(key.getMapParams());
102                 MAP_ADAPTERS.put(key, adapter);
103             } catch (Exception exception) {
104                 throw new CsvEngineException(exception);
105             }
106         }
107         return MAP_ADAPTERS.get(key);
108     }
109 
110     /**
111      * Fill the list of validators.
112      *
113      * @param csvValidations container of the CsvValidation annotation.
114      * @param csvValidation  CsvValidation annotation.
115      */
116     private void constructValidators(CsvValidations csvValidations, CsvValidation csvValidation) {
117         if (csvValidation != null) {
118             validators.add(constructOneValidator(csvValidation));
119         }
120         if (csvValidations != null) {
121             for (CsvValidation oneValidation : csvValidations.value()) {
122                 validators.add(constructOneValidator(oneValidation));
123             }
124         }
125     }
126 
127     /**
128      * Construct one validator (reuse an existing one if the validator with same parameters already exists).
129      *
130      * @param csvValidation CsvValidation annotation.
131      * @return the validator constructed.
132      */
133     private static ValidatorCsv constructOneValidator(CsvValidation csvValidation) {
134         ClassWithParamKey<ValidatorCsv> key =
135                 new ClassWithParamKey<ValidatorCsv>(csvValidation.params(), csvValidation.value());
136         if (!MAP_VALIDATOR.containsKey(key)) {
137             try {
138                 Constructor<? extends ValidatorCsv> construteur =
139                         csvValidation.value().getConstructor((Class<?>[]) null);
140                 ValidatorCsv validator = construteur.newInstance((Object[]) null);
141                 validator.addParams(key.getMapParams());
142                 MAP_VALIDATOR.put(key, validator);
143             } catch (Exception exception) {
144                 throw new CsvEngineException("Error during construction of Validator "
145                         + csvValidation.value().getSimpleName(),
146                         exception);
147             }
148         }
149         return MAP_VALIDATOR.get(key);
150     }

153      * @return field of the class.
154      */
155     public Field getField() {
156         return field;
157     }
158 
159     /**
160      * @return field mandatory?
161      */
162     public boolean isMandatory() {
163         return mandatory;
164     }
165 
166     /**
167      * @return adapter to use.
168      */
169     @SuppressWarnings("unchecked")
170     public AdapterCsv<Object> getAdapterCsv() {
171         return (AdapterCsv<Object>) adapter;
172     }
173 
174     /**
175      * Method which validate the value of a field.
176      *
177      * @param value value to va&lidate.
178      * @throws fr.ybonnel.csvengine.validator.ValidateException throw in case of validation error.
179      */
180     public void validate(String value) throws ValidateException {
181         for (ValidatorCsv validator : validators) {
182             validator.validate(value);
183         }
184     }
